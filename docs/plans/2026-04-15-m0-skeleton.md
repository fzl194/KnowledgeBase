### Task 5: Create run_serving.py entry point

**Files:**
- Create: `agent_serving/scripts/run_serving.py`

**Step 1: Write run_serving.py**

```python
# agent_serving/scripts/run_serving.py
"""Run the agent serving API server.

Usage:
    python -m agent_serving.scripts.run_serving
    python -m agent_serving.scripts.run_serving --port 8080
"""

import argparse


def main() -> None:
    parser = argparse.ArgumentParser(description="Cloud Core Knowledge Backend")
    parser.add_argument("--host", default="127.0.0.1")
    parser.add_argument("--port", type=int, default=8000)
    parser.add_argument("--reload", action="store_true")
    args = parser.parse_args()

    import uvicorn

    uvicorn.run(
        "agent_serving.serving.main:app",
        host=args.host,
        port=args.port,
        reload=args.reload,
    )


if __name__ == "__main__":
    main()
```
